\subsection{实对称矩阵的特征值和特征向量}

\begin{frame}
  \begin{footnotesize}
    \begin{block}{定理}
      实对称矩阵$\A$的任一特征值都是实数。
    \end{block}
    \pause\proofname
    $$
    \A\xx = \lambda \xx
    ~~\Longrightarrow~~
    \overline{(\A\xx)}^T = \overline{\lambda \xx}^T
    ~~\Longrightarrow~~
    \overline{\xx}^T~\overline{\A}^T~\xx = \overline{\lambda} ~\overline{\xx}^T~\xx
    $$
  \end{footnotesize}
\end{frame}


\begin{frame}
  \begin{footnotesize}
    \begin{block}{定理}
      实对称矩阵$\A$对应于不同特征值的特征向量是正交的。
    \end{block}
  \end{footnotesize}
\end{frame}
